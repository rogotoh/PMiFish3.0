USEARCH/VSEARCH Options Used In Each Step

#The values of options marked with * can be changed.
#Please check the USEARCH homepage for details on each option.
#If you want to use an option other than those listed here, please modify the script directly.

##########1_1_Merge_paird_reads##########
(USEARCH/VSEARCH)
-fastq_mergepairs
	*-fastq_trunctail 2	(Default value 2)	only use USEARCH
	*-fastq_minlen 64	(Default value 64)
	*-fastq_maxdiffs 5	(Default value 5)
	*-fastq_pctid 90	(Default value 90)	only use USEARCH
	*-fastq_minovlen 16	(Default value 16)
	 -fastqout

##########1_2_Strip_primers##########
#when "Maxdiff" in Setting.txt is set to "length"
#The VSEARCH does not have this option and The USEARCH12 needs -padlen and -tranclen options, so "Primer_cleaner.pl" is used instead.
(USEARCH)
-fastx_truncate
	 -stripleft N
	 -stripright N
	 N represents the length of each primer specified in Primer.txt
	 -fastqout

##########1_3_Quality_filter##########
(USEARCH/VSEARCH)
-fastq_filter
	*-fastq_maxee 1.0	(Default value 1.0)
	 -fastq_minlen N
	 N corresponds to the "Length = N" value in Setting.txt
	 -fastaout

##########2_1_Find_unique_in_each_sample##########
(USEARCH/VSEARCH)
-fastx_uniques
	 -sizeout
	 -relabel
	*-minuniquesize 1	(Default value 1)
	 -fastaout

##########2_2_Denoise##########
(USEARCH)
-unoise3
	*-unoise_alpha 2.0	(Default value 2.0)
	 -minsize N
	 N corresponds to the "Depth = N" value in Setting.txt
	 -ampout
	 -tabbedout
(VSEARCH)
--cluster_unoise
	*--unoise_alpha 2.0	(Default value 2.0)
	 --minsize N
	 N corresponds to the "Depth = N" value in Setting.txt
	 --centroids

##########2_6_Clustering##########
(USEARCH)
-cluster_smallmem
	 -id N
	 N = V/100, where V is "Cluster_Identity = V" value in Setting.txt 
	 -uc
	 -sortedby size
	 -quiet
(VSEARCH)
--cluster_smallmem
	 --id N
	 N = V/100, where V is "Cluster_Identity = V" value in Setting.txt 
	 --uc
	 --usersort

##########3_1_USEARCH_global##########
(USEARCH/VSEARCH)
-usearch_global
	 -db
	 -id N
	 N = V/100 where V is the "UIdentity = V" value in Setting.txt 
	 -maxaccepts 100	
	 -strand plus
	 -userout
	 -userfields query+target+id+alnlen+mism+opens+qlo+qhi+tlo+thi+qcov

##########4_1_Annotation##########
(USEARCH)
-cluster_smallmem
	 -id N
	 N = V/100, where V is "Cluster_Identity = V" value in Setting.txt 
	 -uc
	 -sortedby size
	 -quiet
(VSEARCH)
--cluster_smallmem
	 --id N
	 N = V/100, where V is "Cluster_Identity = V" value in Setting.txt 
	 --uc
	 --usersort
